declare function={ 
	K(\r) = -1 * exp(-1 * \r) * (1 + \r);
	S(\r) = exp(-1 * \r) * (1 + \r + \r ^2 / 3);
	J(\r) = exp(-2 * \r) * (1 + 1/\r) - 1 / \r;	
	austausch(\r) = - S(\r) / 2 + K(\r) + S(\r) / \r;
	coulomb(\r) = -1/2 + J(\r) + 1/\r;
	up(\r) = (coulomb(\r) + austausch(\r)) / (1 + S(\r)) + 0.5;
	um(\r) = (coulomb(\r) - austausch(\r)) / (1 - S(\r)) + 0.5;
  },
]
