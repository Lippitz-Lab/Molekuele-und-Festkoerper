

\begin{tikzpicture}[
declare function={ 
	k = 1;
	wo = sqrt(k);
	U(\r) = 0.5 * k * \r ^2;	
	E(\n) = wo* (\n + 0.5);
	xi(\r) = \r * k^(1/4);
	pre = (wo / pi)^(1/4);
          H0(\x) = pre * 1;
          H1(\x) = pre / sqrt(2) * 2 * \x;
          H2(\x) = pre / sqrt(8) * (4 * \x^2 - 2);
          H3(\x) =  pre / sqrt(48) *(8 * \x^3 - 12 * \x);
          H4(\x) =  pre / sqrt(384) *(16 * \x^4 - 48 * \x^2 + 12);
          H5(\x) =  pre/ sqrt(3840)  *(32 * \x^5 - 160 * \x^3 + 120 * \x);
          decay(\x) = 0.5 * exp(- \x^2 / 2);
	  },
]
